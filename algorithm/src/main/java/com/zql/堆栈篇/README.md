class Solution {
    public boolean isValid(String s) {
        if (s.length() < 2 || s.length() % 2 != 0) return false;
        ArrayDeque<Character> stack = new ArrayDeque();
        char[] charArr = s.toCharArray();
        for (char ch : charArr) {
            if (stack.isEmpty()) {
                stack.push(ch);
                continue;
            }
            if (('(' == (stack.peek()) && ')' == (ch))
                    || ('{' == (stack.peek()) && '}' == (ch))
                    || ('[' == (stack.peek()) && ']' == (ch))) {
                stack.pop();
                continue;
            } else {
                stack.push(ch);
                continue;
            }
        }
        return stack.isEmpty();
    }
}
